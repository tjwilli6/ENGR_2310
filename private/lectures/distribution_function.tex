\begin{frame}{Expected number of particles in state $s$}
	\begin{align*}
		\left<n_s\right>&=\sum_{R}n_sP(n_s)\\
		\left<n_s\right>&=\frac{\sum_{R}n_se^{-\beta E_R}}{\sum_{R}e^{-\beta E_R}}
	\end{align*}
\end{frame}
